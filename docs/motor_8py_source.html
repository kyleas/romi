36 def set_effort(self, effort):
37 """!
38 @brief Sets the motor effort (speed and direction).
39 @details
40 The effort value must be between -100 and 100. Negative values
41 cause the motor to spin in reverse, while positive values spin forward.
42
43 @exception ValueError If effort is outside the range -100 to 100.
44 @param effort Motor speed setting, from -100 (full reverse)

49 elif effort < 0:
50 self.DIR_pin.high() # Reverse

53
54 self.PWM_pin.pulse_width_percent(abs(effort))
