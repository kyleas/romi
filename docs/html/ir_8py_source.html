24 def __init__(self, ODD, PINS):
25 """!
26 @brief Initializes the IR sensor array.
27 @param ODD Pin controlling the IR LED power (ODD_pin).
28 @param PINS List of 4 pin identifiers used for IR sensor ADC reads.
29 @exception ValueError If PINS does not have exactly 4 elements.
30 """
31 self.ODD_pin = Pin(ODD, mode=Pin.OUT_PP)
32 self.ODD_pin.high()
33
34 if len(PINS) != 4:
35 raise ValueError("PINS list must have exactly 4 elements")
36
37 PIN1_pin = Pin(PINS[0], mode=Pin.IN)
38 PIN3_pin = Pin(PINS[1], mode=Pin.IN)
39 PIN5_pin = Pin(PINS[2], mode=Pin.IN)
40 PIN7_pin = Pin(PINS[3], mode=Pin.IN)
41
42 self.IR1 = pyb.ADC(PIN1_pin)
43 self.IR3 = pyb.ADC(PIN3_pin)
44 self.IR5 = pyb.ADC(PIN5_pin)
45 self.IR7 = pyb.ADC(PIN7_pin)
46
47 self.values = array('f', len(PINS) * [0])
48
49 # Default calibrations
50 self.min1 = 0
51 self.min3 = 0
52 self.min5 = 0
53 self.min7 = 0
54
55 self.max1 = 5.5
56 self.max3 = 5.5
57 self.max5 = 5.5
58 self.max7 = 5.5

60 self.pins = len(PINS)
61
62 def update(self):
63 """!
64 @brief Updates the IR sensor readings, scaling them between 0 and 1.
65 @details
66 Reads each ADC, subtracts the min calibration, and divides by
67 the (max - min) range.
68 """
69 self.values[0] = (self.IR1.read() - self.min1) / (self.max1 - self.min1)
70 self.values[1] = (self.IR3.read() - self.min3) / (self.max3 - self.min3)
71 self.values[2] = (self.IR5.read() - self.min5) / (self.max5 - self.min5)
72 self.values[3] = (self.IR7.read() - self.min7) / (self.max7 - self.min7)
73
74 def calc_centroid(self):
75 """!
76 @brief Computes a centroid-based position indicator from -1 to +1.
77 @details
78 The IR sensors are conceptually at positions [0..3]. Weighted by readings,
79 the centroid is computed. Then scaled to [-1..1].
80
81 @return A float from -1 to +1 representing the line's position under the sensors.
82 """
83 self.update()
84 sum_values = 0
85 moment = 0
86 for i in range(self.pins):
87 moment += self.values[i] * i
88 sum_values += self.values[i]
89
90 if sum_values > 0:
91 return moment / sum_values / ((self.pins - 1) / 2) - 1
92 return 0
