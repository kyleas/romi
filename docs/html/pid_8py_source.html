43 def update(self, desire, actual):
44 """!
45 @brief Calculates a control effort using a PID approach (plus feedforward).
46 @details
47 Uses the difference between desired and actual values (error) to compute the
48 PID output. Limits integral error to prevent excessive windup.
49 @param desire The target or desired value.
50 @param actual The current or measured value.
51 @return The control effort as a float (could be motor PWM, for example).
52 """
53 dt = ticks_diff(ticks_ms(), self.prev_time) / 1000
54 self.prev_time = ticks_ms()
55
56 error = desire - actual
57 self.integral_error += error * dt
58 # Bound integrator
59 if self.integral_error > self.integral_error_max:
60 self.integral_error = self.integral_error_max
61 elif self.integral_error < -self.integral_error_max:
62 self.integral_error = -self.integral_error_max
63
64 derivative = error - self.prev_error
65 self.prev_error = error
66
67 # Determine direction for feedforward offset
68 dir = 1 if desire > 0 else -1 if desire < 0 else 0
69
70 return (self.Kp * error
71 + self.integral_error * self.Ki
72 + self.Kd * derivative
73 + self.ff_gain * desire
74 + (self.ff_offset * dir))
