156STATIC mp_obj_t IntQueue_put(mp_obj_t self_in, mp_obj_t to_put)
157{
158 cqueue_IntQueue_obj_t *self = MP_OBJ_TO_PTR(self_in);
159 int32_t putted = mp_obj_get_int(to_put);
160
161 self->p_data[self->write_idx] = putted;
162 self->write_idx++;
163 if (self->write_idx >= self->size)
164 {
165 self->write_idx = 0;
166 }
167
168 // If the queue is full before writing, move the read pointer so we'll read
169 // old data, not new data
170 if (self->num_items >= self->size)
171 {
172 self->read_idx++;
173 if (self->read_idx >= self->size)
174 {
175 self->read_idx = 0;
176 }
177 }
178
179 // Now increase the fillage and check again if the queue is full
180 self->num_items++;
181 if (self->num_items >= self->size)
182 {
183 self->num_items = self->size;
184 }
185 if (self->num_items > self->max_full)
186 {
187 self->max_full = self->num_items;
188 }
189
190 return mp_const_none;

638STATIC mp_obj_t ByteQueue_put(mp_obj_t self_in, mp_obj_t str_obj_in)
639{
640 // Ensure that the input is a valid type (prevents crashes)
641 if (!mp_obj_is_str_or_bytes(str_obj_in))
642 {
643 mp_raise_TypeError("Bytes or string required");
644 }
645
646 cqueue_ByteQueue_obj_t *self = MP_OBJ_TO_PTR(self_in);
647
648 // Get the length of the bytearray or string and a data pointer
649 GET_STR_DATA_LEN(str_obj_in, my_str, str_len);
650
651 // Copy the data into the queue, overwriting old data if it's there
652 for (size_t index = 0; index < str_len; index++)
653 {
654 self->p_data[self->write_idx] = my_str[index];
655 self->write_idx++;
656 if (self->write_idx >= self->size)
657 {
658 self->write_idx = 0;
659 }
660 if (self->num_items >= self->size) // If the queue is full before
661 { // writing, move read pointer
662 self->read_idx++; // so we'll read old data,
663 if (self->read_idx >= self->size) // not new data
664 {
665 self->read_idx = 0;
666 }
667 }
668 self->num_items++; // Now increase the fillage and
669 if (self->num_items >= self->size) // check again if the queue is
670 { // full
671 self->num_items = self->size;
672 }
673 if (self->num_items > self->max_full)
674 {
675 self->max_full = self->num_items;
676 }
677 }
678
679 return mp_const_none;
680}
